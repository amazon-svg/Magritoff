  </div>


  <script>

    const chatContainer = document.getElementById('chat-container');

    const messageInput = document.getElementById('message-input');

    const sendButton = document.getElementById('send-button');


    function addMessage(content, type) {

      const messageDiv = document.createElement('div');

      messageDiv.className = `message ${type}-message`;

      messageDiv.textContent = content;

      chatContainer.appendChild(messageDiv);

      chatContainer.scrollTop = chatContainer.scrollHeight;

    }


    function sendMessage() {

      const message = messageInput.value.trim();

      if (!message) return;


      addMessage(message, 'user');

      messageInput.value = '';

      sendButton.disabled = true;


      addMessage('Claude réfléchit...', 'loading');


      parent.postMessage({

        pluginMessage: {

          type: 'send-message',

          message: message

        }

      }, '*');

    }


    sendButton.addEventListener('click', sendMessage);

    messageInput.addEventListener('keypress', (e) => {

      if (e.key === 'Enter') sendMessage();

    });


    window.onmessage = (event) => {

      const msg = event.data.pluginMessage;

      

      // Remove loading message

      const loadingMessages = chatContainer.querySelectorAll('.loading');

      loadingMessages.forEach(el => el.remove());


      if (msg.type === 'assistant-response') {

        addMessage(msg.message, 'assistant');

      } else if (msg.type === 'error') {

        addMessage(`Erreur: ${msg.message}`, 'error');

      }


      sendButton.disabled = false;

      messageInput.focus();

    };

  </script>
